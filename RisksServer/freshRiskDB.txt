-- SQL command used to create risk table within database
CREATE TABLE IF NOT EXISTS riskEvent (
	rID INTEGER,
	rName VARCHAR(250) NOT NULL,
	impact INTEGER,
	probability INTEGER,
	description TEXT,
	mitigation TEXT,
	status VARCHAR(25),
	project VARCHAR(250) NOT NULL,
	FOREIGN KEY (project) REFERENCES project(pName) ON DELETE CASCADE ON UPDATE CASCADE,
	CONSTRAINT constraint_1 PRIMARY KEY(rName,project)
);
